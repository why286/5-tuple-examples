##count() - Returns the number of times a specified value appears in the tuple. t = (1, 2, 3, 2, 4, 2) print(t.count(2)) ##index() - Returns the index of the first occurrence of the specified value. t = (1, 2, 3, 4, 5) print(t.index(3)) ##len() - Returns the number of elements in the tuple t = (1, 2, 3, 4, 5) print(len(t)) ##max() - Returns the largest item in the tuple. t = (1, 2, 3, 4, 5) print(max(t)) ##min() - Returns the smallest item in the tuple. t = (1, 2, 3, 4, 5) print(min(t))
